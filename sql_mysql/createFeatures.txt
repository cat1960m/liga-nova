CREATE TABLE text_contents (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    text_description_id INT, 
    language TEXT, 
    text_content TEXT, 
    updated_by INT, 
    last_edit_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, 
    content_type TEXT,
    CONSTRAINT fk_text_description FOREIGN KEY (text_description_id) REFERENCES text_descriptions(id) ON DELETE CASCADE ON UPDATE CASCADE
);
//////////

CREATE TABLE text_descriptions (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    feature_id INT, 
    text_type VARCHAR(256) NOT NULL, 
    price INT, 
    can_delete BOOL, 
    text_description_order BIGINT, 
    value TEXT, 
    link TEXT,
    CONSTRAINT fk_feature FOREIGN KEY (feature_id) REFERENCES features(id) ON DELETE CASCADE ON UPDATE CASCADE
);
